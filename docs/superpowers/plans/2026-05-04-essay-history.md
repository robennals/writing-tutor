# Essay History Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Capture a snapshot of the draft each time the student presses "Check My Writing" or "I've Made Changes!", link it to the corresponding chat message, and let anyone view past versions as a word-level diff (bold green for additions, faint red for deletions) with prev/next navigation and a versions button.

**Architecture:** A new `essay_snapshots` table plus a `snapshot_id` column on `messages`. The two button handlers in `WritingScreen` POST to a new `/api/essays/:id/snapshots` route before sending the chat message; the chat route reads the resulting `snapshotId` and persists it on the user message. On the client, `activeSnapshotId` state replaces the editor with a read-only diff view, highlights the matching chat bubble, and is set/cleared by inline links, prev/next buttons, a versions button, and Escape.

**Tech Stack:** Next.js 16 App Router, React 19, TypeScript, libsql (`@libsql/client`), Vitest + React Testing Library, Tailwind. New runtime dep: `diff` (jsdiff) for word-level diffing.

**Package manager:** pnpm. Run tests with `pnpm test`, lint with `pnpm lint`, type-check with `pnpm exec tsc --noEmit`.

**Spec:** `docs/superpowers/specs/2026-05-04-essay-history-design.md`

```ts
export async function createSnapshot(
  essayId: number,
  content: string
): Promise<number> {
  const result = await db.execute({
    sql: "INSERT INTO essay_snapshots (essay_id, content) VALUES (?, ?)",
    args: [essayId, content],
  });
  return Number(result.lastInsertRowid);
}

export async function getSnapshots(essayId: number): Promise<EssaySnapshot[]> {
  const result = await db.execute({
    sql: "SELECT * FROM essay_snapshots WHERE essay_id = ? ORDER BY created_at ASC, id ASC",
    args: [essayId],
  });
  return result.rows.map((r) => toPlain<EssaySnapshot>(r));
}
```

```ts
import { getSession } from "@/lib/auth";
import { initializeDatabase } from "@/lib/db-schema";
import { createSnapshot, getSnapshots } from "@/lib/queries";
import { NextRequest, NextResponse } from "next/server";

let dbInitialized = false;

async function ensureDb() {
  if (!dbInitialized) {
    await initializeDatabase();
    dbInitialized = true;
  }
}

export async function POST(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const session = await getSession();
  if (!session || session.role !== "child") {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  await ensureDb();
  const { id } = await params;
  const { content } = (await req.json()) as { content: string };
  const snapshotId = await createSnapshot(Number(id), content);
  const [snap] = await getSnapshots(Number(id)).then((rows) =>
    rows.filter((r) => r.id === snapshotId)
  );
  return NextResponse.json({ id: snapshotId, created_at: snap.created_at });
}

export async function GET(
  _req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const session = await getSession();
  if (!session) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  await ensureDb();
  const { id } = await params;
  const snapshots = await getSnapshots(Number(id));
  return NextResponse.json({ snapshots });
}
```

```tsx
"use client";

import { diffWordsWithSpace } from "diff";
import { htmlToPlainText } from "@/lib/utils";

function normalizeWithParagraphs(html: string): string {
  // htmlToPlainText collapses to one line. Re-introduce paragraph breaks so
  // the rendered diff matches the editor's structure.
  return html
    .replace(/<\/p>\s*<p[^>]*>/gi, "\n\n")
    .replace(/<p[^>]*>/gi, "")
    .replace(/<\/p>/gi, "")
    .replace(/<br\s*\/?>/gi, "\n")
    .replace(/<[^>]+>/g, "")
    .replace(/&nbsp;/g, " ")
    .replace(/&amp;/g, "&")
    .replace(/&lt;/g, "<")
    .replace(/&gt;/g, ">")
    .trim();
}

interface Segment {
  text: string;
  kind: "same" | "added" | "removed";
}

function paragraphs(segments: Segment[]): Segment[][] {
  // Split each segment on \n\n boundaries; each resulting chunk is its own
  // segment. Then group into paragraphs by walking until we cross an empty
  // boundary in the original stream.
  const out: Segment[][] = [[]];
  for (const seg of segments) {
    const parts = seg.text.split(/\n\n+/);
    parts.forEach((part, i) => {
      if (i > 0) out.push([]);
      if (part.length > 0) out[out.length - 1].push({ ...seg, text: part });
    });
  }
  return out.filter((p) => p.length > 0);
}

export function DiffView({
  prev,
  current,
}: {
  prev: string | null;
  current: string;
}) {
  const currentText = normalizeWithParagraphs(current);

  if (prev === null) {
    return (
      <div className="px-5 py-4 text-base leading-[1.8] space-y-3">
        <p className="text-xs text-muted-foreground italic">(first version)</p>
        {currentText.split(/\n\n+/).map((para, i) => (
          <p key={i}>{para}</p>
        ))}
      </div>
    );
  }

  const prevText = normalizeWithParagraphs(prev);
  if (prevText === currentText) {
    return (
      <div className="px-5 py-4 text-base leading-[1.8] space-y-3">
        <p className="text-xs text-muted-foreground italic">
          (no edits between this version and the previous one)
        </p>
        {currentText.split(/\n\n+/).map((para, i) => (
          <p key={i}>{para}</p>
        ))}
      </div>
    );
  }

  const raw = diffWordsWithSpace(prevText, currentText);
  const segments: Segment[] = raw.map((c) => ({
    text: c.value,
    kind: c.added ? "added" : c.removed ? "removed" : "same",
  }));

  return (
    <div className="px-5 py-4 text-base leading-[1.8] space-y-3">
      {paragraphs(segments).map((para, pi) => (
        <p key={pi}>
          {para.map((seg, si) => {
            if (seg.kind === "added") {
              return (
                <span
                  key={si}
                  data-diff="added"
                  className="font-bold underline decoration-2 decoration-emerald-400 text-emerald-300"
                >
                  {seg.text}
                </span>
              );
            }
            if (seg.kind === "removed") {
              return (
                <span
                  key={si}
                  data-diff="removed"
                  className="line-through decoration-2 decoration-rose-400 text-rose-400/50"
                >
                  {seg.text}
                </span>
              );
            }
            return <span key={si}>{seg.text}</span>;
          })}
        </p>
      ))}
    </div>
  );
}
```

```tsx
function HistoryView({
  snapshots,
  activeSnapshotId,
  onPrev,
  onNext,
  onClose,
}: {
  snapshots: EssaySnapshot[];
  activeSnapshotId: number;
  onPrev: () => void;
  onNext: () => void;
  onClose: () => void;
}) {
  const idx = snapshots.findIndex((s) => s.id === activeSnapshotId);
  const current = snapshots[idx];
  const previous = idx > 0 ? snapshots[idx - 1] : null;
  const time = current
    ? new Date(current.created_at).toLocaleTimeString([], {
        hour: "numeric",
        minute: "2-digit",
      })
    : "";

  return (
    <div className="flex-1 flex flex-col min-h-0">
      <div className="mx-5 mt-3 mb-2 px-3 py-2 rounded-lg bg-amber-950/30 border border-amber-600/40 flex items-center justify-between shrink-0">
        <div className="text-sm text-amber-200">

          <span className="text-xs text-amber-300/70">
            ({idx + 1} of {snapshots.length})
          </span>
        </div>
        <div className="flex items-center gap-1">
```

## Notes on production deploy

The migration runs automatically on the first request to any route after deploy (the existing `dbInitialized` flag pattern in every route file). Running concurrently across cold-starting Vercel function instances is safe: `CREATE TABLE IF NOT EXISTS` is a no-op on the second runner; the `ALTER TABLE messages ADD COLUMN` either succeeds once or throws "duplicate column," which the existing try/catch in `db-schema.ts` swallows. No manual migration step required.
